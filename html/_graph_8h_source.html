181 * Auxiliary function to add an outgoing edge to a vertex (this),
182 * with a given destination vertex (d) and edge weight (w).

184template <class T>
185Edge<T> * Vertex<T>::addEdge(Vertex<T> *d, int driving, int walking) {
186 auto newEdge = new Edge<T>(this, d, driving, walking);
187 adj.push_back(newEdge);
188 d->incoming.push_back(newEdge);
189 return newEdge;
190}
191
192/*
193 * Auxiliary function to remove an outgoing edge (with a given destination (d))
194 * from a vertex (this).
195 * Returns true if successful, and false if such edge does not exist.
196 */
197template <class T>
198bool Vertex<T>::removeEdge(T in) {
199 bool removedEdge = false;
200 auto it = adj.begin();
201 while (it != adj.end()) {
202 Edge<T> *edge = *it;
203 Vertex<T> *dest = edge->getDest();
204 if (dest->getId() == in) {
205 it = adj.erase(it);
206 deleteEdge(edge);
207 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
208 }
209 else {
210 it++;
211 }
212 }
213 return removedEdge;
214}

216/*
217 * Auxiliary function to remove an outgoing edge of a vertex.
218 */
219template <class T>
220void Vertex<T>::removeOutgoingEdges() {
221 auto it = adj.begin();
222 while (it != adj.end()) {
223 Edge<T> *edge = *it;
224 it = adj.erase(it);
225 deleteEdge(edge);
226 }
227}

349template <class T>
350void Vertex<T>::deleteEdge(Edge<T> *edge) {
351 Vertex<T> *dest = edge->getDest();
352 // Remove the corresponding edge from the incoming list
353 auto it = dest->incoming.begin();
354 while (it != dest->incoming.end()) {
355 if ((*it)->getOrig()->getId() == id) {
356 it = dest->incoming.erase(it);
357 }
358 else {
359 it++;
360 }
361 }
362 delete edge;
363}

476/*
477 * Removes a vertex with a given content (in) from a graph (this), and
478 * all outgoing and incoming edges.
479 * Returns true if successful, and false if such vertex does not exist.
480 */
481template <class T>
482bool Graph<T>::removeVertex(const T &in) {
483 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
484 if ((*it)->getId() == in) {
485 auto v = *it;
486 v->removeOutgoingEdges();
487 for (auto u : vertexSet) {
488 u->removeEdge(v->getId());
489 }
490 vertexSet.erase(it);
491 delete v;
492 return true;
493 }
494 }
495 return false;
496}

503template <class T>

527template <class T>
528bool Graph<T>::addBidirectionalEdge(const std::string &sourc, const std::string &dest, int drivingTime, int walkingTime) {
529 auto v1 = findVertex(sourc);
530 auto v2 = findVertex(dest);
531 if (v1 == nullptr || v2 == nullptr)
532 return false;
533 auto e1 = v1->addEdge(v2, drivingTime, walkingTime);
534 auto e2 = v2->addEdge(v1, drivingTime, walkingTime);
535 e1->setReverse(e2);
536 e2->setReverse(e1);
537 return true;
538}
